Raphaël JS 
* JavaScript library (obvious!)
* browser based graphics drawing 
* vector graphics 
* **but not** (explicitly) an SVG library
	  
Raphaël JS: 
* vector based 
* shapes are part of the DOM tree 
* interaction with elements via DOM (events, ...) 

HTML5 Canvas: 
* bitmap based
* single DOM element (the "canvas"-tag)
* no DOM-style interaction
		
Raphaël JS: 
* wide browser support (FF3+, WebKit3+, **IE6+**)
* SVG engine in FF, WebKit, IE9
* VML engine in IE6-8

HTML5 Canvas:
* only supported by new browsers (IE: 9+)
		
	 HTML:
	 
<div id="example1" style="border: 1px solid #000; width: 500px; height: 500px;">
  <!-- here goes the graphics -->

paper.circle(250, 250, 50);
paper.ellipse(50, 350, 100, 50);
paper.rect(150, 100, 200, 50);
paper.rect(350, 250, 100, 50, 6); // rounded corners
paper.path("M50 50 L100 50 L50 100 L50 50");
paper.text(200, 200, "Some text");
Setting shape/element attributes: {{{attr()}}}
// solid fill, with opacity
paper.circle(250, 250, 50).attr({
  "stroke-width" : 3, "fill" : "#000", "opacity" : 0.5
});

// radial gradient fill
paper.ellipse(50, 350, 100, 50).attr({
  "fill" : "r(0.25, 0.75)#fff-#000" 
});

// linear gradient fill
paper.rect(150, 100, 200, 50).attr({
  "stroke-width" : 3, "fill" : "90-#ff0000-#fff-#0000ff"
});

// dashed stroke 
paper.path("M50 50 L100 50 L50 100 L50 50").attr({
  "stroke-dasharray" : "- ", "fill" : "#0000ff" 
});
	  
Animating shapes
* using the {{{animate(...)}}}-method
* arguments: new attributes, duration, easing-formula and optional callback function
var circle = paper.circle(250, 250, 50).attr({"fill" : "#506090"});
// ...
circle.animate({ cx: 100, cy: 100, r : 10}, 500, "linear");
// ...
circle.animate({ cx: 400, cy: 400, r : 90}, 500, "bounce");
// ...
circle.animate({ cx: 250, cy: 250, r : 50}, 500, "elastic", function() {
  alert("Back to center!");
});

Attach event handlers
* every shape is a DOM elements and thus supports DOM events 
* the DOM node is accessible via the {{{.node}}}-property
* (Raphael also provides it's own methods to attach event handlers)
var circle = paper.circle(250, 250, 50).attr({"fill" : "#506090"});

// attach a handler (JQuery)
$(circle.node).bind("mouseover", function() {
  var nx = Math.random() * 500;
  var ny = Math.random() * 500;
  circle.animate({ cx : nx, cy : ny}, 500, "bounce"); 
});
    
Add custom methods to Raphael canvas:

Raphael.fn.star = function(cx, cy, ir, or, n) {
  var a = Math.PI / n;
  var sa = 0; // (Math.PI / 3);
  var path = "M" + (cx) + " " + (cy + or) + " ";
  for ( var i = 0; i < n; i++) {
    var a1 = sa + (a * (i * 2));
    var a2 = sa + (a * (i * 2 + 1));
    path += "L" + (cx + Math.sin(a1) * or) + " " + (cy + Math.cos(a1) * or) + " ";
    path += "L" + (cx + Math.sin(a2) * ir) + " " + (cy + Math.cos(a2) * ir) + " ";
  }
  path += "L" + (cx) + " " + (cy + or) + " ";
  return this.path(path);
}

// now draw some stars...
var paper = Raphael('paper');
paper.star(300, 300, 20, 40, 7).attr({ "stroke" : "#a0a000" });
paper.star(100, 300, 5, 60, 6).attr({ "fill" : "#40d0ff" });
paper.star(350, 150, 50, 60, 50).attr({ "fill" : "#ffff00", "stroke" : "#ff0000" });
    
Some pit falls...
* grouping of elements (i.e. svg:g element) not supported
* drawing of complex graphics with thousands of elements may get slow (IE6-8!)
* mixing with plain SVG images not directly supported (IE compatibility)
